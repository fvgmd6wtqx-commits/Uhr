<script>

let startReal;
let timer;

function updateClockType(){

let type=document.getElementById("clockType").value;

digitalClock.style.display="none";
analogClock.style.display="none";
circleClock.style.display="none";

if(type==="digital") digitalClock.style.display="block";
if(type==="analog") analogClock.style.display="block";
if(type==="circle") circleClock.style.display="block";

}

document.getElementById("clockType").addEventListener("change",updateClockType);

function startSimulation(){

let startInput=startTime.value;
let endInput=endTime.value;
let duration=parseFloat(duration.value);

let s=startInput.split(":");
let e=endInput.split(":");

let startMinutes=s[0]*60 + parseInt(s[1]);
let endMinutes=e[0]*60 + parseInt(e[1]);

let simLength=endMinutes-startMinutes;
let factor=simLength/duration;

startReal=Date.now();

timer=setInterval(()=>{

let elapsed=(Date.now()-startReal)/60000;
let simMinutes=startMinutes + elapsed*factor;

let progress=elapsed/duration;

if(simMinutes>=endMinutes){
simMinutes=endMinutes;
progress=1;
clearInterval(timer);
}

let h=Math.floor(simMinutes/60);
let m=Math.floor(simMinutes%60);

let timeStr=
String(h).padStart(2,"0")+":"+String(m).padStart(2,"0");

digitalClock.innerText=timeStr;

drawAnalog(h,m);
drawCircle(progress);

},1000);

}

function resetSimulation(){
clearInterval(timer);
digitalClock.innerText=startTime.value;
drawCircle(0);
}

function drawAnalog(h,m){

let canvas=document.getElementById("analogClock");
let ctx=canvas.getContext("2d");

ctx.clearRect(0,0,200,200);

ctx.beginPath();
ctx.arc(100,100,90,0,Math.PI*2);
ctx.strokeStyle="white";
ctx.stroke();

let hourAngle=(h%12 + m/60)*30*Math.PI/180;
let minuteAngle=m*6*Math.PI/180;

ctx.beginPath();
ctx.moveTo(100,100);
ctx.lineTo(
100+50*Math.sin(hourAngle),
100-50*Math.cos(hourAngle)
);
ctx.stroke();

ctx.beginPath();
ctx.moveTo(100,100);
ctx.lineTo(
100+70*Math.sin(minuteAngle),
100-70*Math.cos(minuteAngle)
);
ctx.stroke();

}

function drawCircle(p){

let deg=p*360;
circleFill.style.background=
`conic-gradient(#4CAF50 ${deg}deg,#333 ${deg}deg)`;

}

updateClockType();

</script>
